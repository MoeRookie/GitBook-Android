public class XmlParserUtils {
    public static final String CHANNEL = "channel";
    public static final String ITEM = "item";
    public static final String TITLE = "title";
    public static final String DESC = "description";
    public static final String IMG_URL = "image";
    public static final String TYPE = "type";
    public static final String COMMENT = "comment";

    /**
     * 解析输入流为新闻集合
     * @param inputStream 输入流
     * @return 新闻集合
     */
    public static List<NewsBean> parse(InputStream inputStream) throws Exception{
        List<NewsBean> newsList = null;
        NewsBean news = null;
        // [1]获取xml解析器
        XmlPullParser parser = Xml.newPullParser();
        // [2]设置解析器要解析的内容
        parser.setInput(inputStream, "utf-8");
        // [3]获取解析的事件类型
        int eventType = parser.getEventType();
        // [4]不停的向下解析
        while (eventType != XmlPullParser.END_DOCUMENT) {
            // [5]判断解析的是开始节点还是结束节点
            switch (eventType) {
                case XmlPullParser.START_TAG:
                    // [6]判断解析的是哪个开始标签
                    if (CHANNEL.equals(parser.getName())) {
                        // 创建newsList集合
                        newsList = new ArrayList<>();
                    } else if (ITEM.equals(parser.getName())) {
                        // 创建news对象
                        news = new NewsBean();
                    } else if (TITLE.equals(parser.getName())) {
                        String title = parser.nextText();
                        news.setTitle(title);
                    } else if (DESC.equals(parser.getName())) {
                        String desc = parser.nextText();
                        news.setDesc(desc);
                    } else if (IMG_URL.equals(parser.getName())) {
                        String imgUrl = parser.nextText();
                        news.setImgUrl(imgUrl);
                    } else if (TYPE.equals(parser.getName())) {
                        String type = parser.nextText();
                        news.setType(Integer.parseInt(type));
                    } else if (COMMENT.equals(parser.getName())) {
                        String comment = parser.nextText();
                        news.setComment(Integer.parseInt(comment));
                    }
                    break;
                case XmlPullParser.END_TAG:
                    // 添加news到newsList中
                    newsList.add(news);
                    break;
            }
            eventType = parser.next();
        }
        return newsList;
    }
}

                super.run();
                try {
                    // [2.1]创建URL对象

                    // [2.3]设置发送get请求

                    // [2.4]设置请求超时时间
                    conn.setConnectTimeout(5000);
                    // [2.5]获取服务器返回的状态码
                    int code = conn.getResponseCode();
                    if (200 == code) {

                    e.printStackTrace();
